FROM #table2

;

WITH CTE_LEAD AS
(
SELECT visit_date AS startDate
	, LEAD(visit_date) OVER(ORDER BY visit_date) AS endDate
FROM #table2
)
SELECT DATEADD(dd, 1, startDate) AS startDate
	,DATEADD(dd, -1, endDate) endDate
FROM CTE_LEAD
WHERE DATEDIFF(dd, startDate, endDate) > 1
;
